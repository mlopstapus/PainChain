Your release is named {{ .Release.Name }}.

To learn more about the release, try:

  $ helm status {{ .Release.Name }}
  $ helm get all {{ .Release.Name }}

To access your application:

{{- if eq .Values.frontend.service.type "NodePort" }}
1. Get the frontend URL by running:


  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "painchain.fullname" . }}-frontend)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT

{{- else if eq .Values.frontend.service.type "LoadBalancer" }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "painchain.fullname" . }}-frontend --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo http://$SERVICE_IP:{{ .Values.frontend.service.port }}

{{- else if eq .Values.frontend.service.type "ClusterIP" }}
  # Terminal 1: Forward the backend API
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "painchain.fullname" . }}-backend {{ .Values.backend.service.port }}:{{ .Values.backend.service.port }}

  # Terminal 2: Forward the frontend
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "painchain.fullname" . }}-frontend {{ .Values.frontend.service.port }}:{{ .Values.frontend.service.port }}

  echo "Visit http://127.0.0.1:{{ .Values.frontend.service.port }} to use your application"
  echo "Backend API available at http://127.0.0.1:{{ .Values.backend.service.port }}"
{{- end }}

IMPORTANT: You must run BOTH port-forward commands above (in separate terminals) for the application to work properly.
The frontend makes API calls to the backend at http://localhost:{{ .Values.backend.service.port }}
